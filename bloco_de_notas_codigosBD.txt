CREATE TABLE [dbo].[moradores]
(
	[Cod_morador] INT NOT NULL PRIMARY KEY, 
    [Nome] NCHAR(10) NULL, 
    [CPF] NCHAR(10) NULL, 
    [Rua] NCHAR(10) NULL, 
    [Unidade] NCHAR(10) NULL, 
    [DDD_fixo] NCHAR(10) NULL, 
    [Telefone_fixo] NCHAR(10) NULL, 
    [DDD_celular] NCHAR(10) NULL, 
    [Telefone_celular] NCHAR(10) NULL, 
    [Email] NCHAR(10) NULL, 
    [Observações] NCHAR(10) NULL
)
tabela de usuarios
CREATE TABLE [dbo].[usuarios] (
    [Cod]     INT          NOT NULL,
    [usuario] VARCHAR (10) NOT NULL,
    [senha]   VARCHAR (10) NOT NULL,
    PRIMARY KEY CLUSTERED ([Cod] ASC)
);
tabela de visitantes
CREATE TABLE [dbo].[visitantes] (
    [Cod_morador]      INT           NOT NULL,
    [Nome]             VARCHAR (50)  NOT NULL,
    [CPF]              INT           NOT NULL,
    [visitado]         VARCHAR (50)  NOT NULL,
    [Unidade]          INT           NOT NULL,
    [DDD_fixo]         INT           NULL,
    [Telefone_fixo]    INT           NULL,
    [DDD_celular]      INT           NOT NULL,
    [Telefone_celular] INT           NOT NULL,
    [Email]            VARCHAR (50)  NULL,
    [Observações]      VARCHAR (250) NULL,
    [foto]             IMAGE         NULL,
    [marca]            VARCHAR (15)  NULL,
    [modelo]           VARCHAR (15)  NULL,
    [cor]              VARCHAR (15)  NULL,
    [ano]              DATE          NULL,
    [placa]            VARCHAR (8)   NULL,
    PRIMARY KEY CLUSTERED ([Cod_morador] ASC)
);
